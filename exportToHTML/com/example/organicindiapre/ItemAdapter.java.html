17   /*  Aditya Kumar 
18       Used for the Outermost card with Customer and their product details. 
19       Binds one customer to one Holder. 
20       For layout of the card, check customer_row_0.xmlml 
21    */
22   public class ItemAdapter extends RecyclerView.Adapter<ItemAdapter.ItemViewHolder> {
23   
24       private RecyclerView.RecycledViewPool viewPool = new RecyclerView.RecycledViewPool();
25       private List<CustomerClass> itemList;
26   
27       public ItemAdapter(List<CustomerClass> itemList) {
28           this.itemList = itemList;
29       }
30       @NonNull
31       @Override
32       public ItemViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
33           View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.customer_row_0, parent, false);
34           return new ItemViewHolder(view);
35       }
36   
37       @Override
38       public void onBindViewHolder(@NonNull ItemViewHolder holder, int position) {
39           CustomerClass customerObj =itemList.get(position);
40           holder.cust_name.setText(customerObj.getCust_name());
41           holder.cust_phone_no.setText(customerObj.getPhone_no());
42           holder.cust_address.setText(customerObj.getAddress());
43           LinearLayoutManager layoutManager = new LinearLayoutManager(
44                   holder.productRecyclerView.getContext(),
45                   LinearLayoutManager.VERTICAL,
46                   false
47           );
48           layoutManager.setInitialPrefetchItemCount(customerObj.getProductList().size());
49           SubItemAdapter subItemAdapter = new SubItemAdapter(customerObj.getProductList());
50           holder.productRecyclerView.setLayoutManager(layoutManager);
51           holder.productRecyclerView.setAdapter(subItemAdapter);
52           holder.productRecyclerView.setRecycledViewPool(viewPool);
53   
54       }
55   
56       @Override
57       public int getItemCount() {
58           return itemList.size();
59       }
60   
61       static class ItemViewHolder extends RecyclerView.ViewHolder {
62           private TextView cust_name;
63           private TextView cust_phone_no;
64           private  TextView cust_address;
65           private RecyclerView productRecyclerView;
66           ItemViewHolder(@NonNull View itemView) {
67               super(itemView);
68               cust_name = itemView.findViewById(R.id.customer_name);
69               cust_phone_no = itemView.findViewById(R.id.phone_no);
70               cust_address = itemView.findViewById(R.id.address);
71               productRecyclerView = itemView.findViewById(R.id.recycViewProductList);
72           }
73       }
74   }
