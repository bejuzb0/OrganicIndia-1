93       @RequiresApi(api = Build.VERSION_CODES.KITKAT)
94       private void showSelected(final String title, final ProductsAdapters adapter)
95       {
96           if (adapter.getSelectedList() != null)
97           {
98               int arraySize = adapter.getSelectedList().size();
99               if (arraySize > 0)
100              {
101                  if (arraySize  < 4 )
102                  {
103                      final String name= String.valueOf(adapter.getSelectedList().get(0));
104                      final String quantity = String.valueOf(adapter.getSelectedList().get(1));
105                      final String price = String.valueOf(adapter.getSelectedList().get(2));
106                  new MaterialAlertDialogBuilder(requireContext())
107                          .setTitle(title)
108                          .setMessage("SELECTED ITEMS (" + arraySize + ") : \n"+adapter.getSelectedList().toString())
109                          .setPositiveButton("Yes", new DialogInterface.OnClickListener() {
110                              @Override
111                              public void onClick(DialogInterface dialog, int which)
112                              {
113                                  Intent intent = new Intent(getContext(),Delivery.class);
114  
115                                  //passing data from fragment to delivery activity
116                                  intent.putExtra("PN",name);
117                                  intent.putExtra("PQ",quantity);
118                                  intent.putExtra("PP",price);
119                                  //one time or subscription type
120                                  intent.putExtra("Type",title);
121                                  startActivity(intent);
122                              }
123                          }).setNegativeButton("No", new DialogInterface.OnClickListener() {
124                      @Override
125                      public void onClick(DialogInterface dialog, int which) {
126                          dialog.dismiss();
127                      }
128                  }).show();
129                  }
130                  else
131                  {
132                      Toast.makeText(getActivity(), "Select One product at a time only", Toast.LENGTH_SHORT).show();
133                  }
134  
135              }
136              else {
137                  Toast.makeText(getActivity(), "Select Atleast One ", Toast.LENGTH_SHORT).show();
138              }
139          }
140          else {
141              Toast.makeText(getActivity(), "Select Atleast One ", Toast.LENGTH_SHORT).show();
142          }
143      }
